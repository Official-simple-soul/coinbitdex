The issue here is that TextInput does not directly accept a width prop to control the width of the input field. Instead, you should use a wrapper component like Box or a style prop with flex properties in the Group component to control the layout. Below is the corrected implementation:

jsx
<Group grow mt={14} align="flex-start" style={{ gap: '10px' }}>
  <TextInput
    label="Phone Prefix"
    value={form.values.phonePrefix}
    readOnly
    style={{ flex: '0 0 20%' }} // Adjust width for the prefix input
  />
  <TextInput
    label="Phone"
    placeholder="Phone Number"
    {...form.getInputProps('phone')}
    required
    style={{ flex: '1' }} // Adjust width for the phone input
  />
</Group>

Explanation:

  1. style Prop with flex: The flex property ensures that the Phone Prefix input takes up 20% of the available width, while the Phone input takes the remaining space (flex: 1).
  2. gap: Adds spacing between the inputs for better UI separation.
  3. align="flex-start": Ensures the inputs are aligned properly within the group.

This approach ensures that both inputs do not share the same width and that the layout is responsive. Let me know if you need further customization!

import { useForm } from '@mantine/form';
import {
TextInput,
PasswordInput,
Button,
Modal,
} from '@mantine/core';
import { useState } from 'react';
import { useNavigate } from 'react-router';

function Login() {
const [opened, setOpened] = useState(true);

const form = useForm({
initialValues: {
email: '',
password: '',
},

validate: {
email: (value) => (/^\S+@\S+\.\S+$/.test(value) ? null : 'Invalid email'),
password: (value) => (value.length >= 6 ? null : 'Password must be at least 6 characters'),
},
});

const navigate = useNavigate();
const close = () => {
// Navigate to home page
navigate('/');
};

return (
<div className="flex min-h-screen w-full items-center justify-center bg-auth-bg bg-no-repeat bg-cover bg-center">
<Modal opened={opened} onClose={close} title="Login" centered>
<form onSubmit={form.onSubmit((values) => console.log(values))}>
<TextInput
placeholder="Email"
{...form.getInputProps('email')}
required
mt={14}
/>

<PasswordInput
placeholder="Password"
{...form.getInputProps('password')}
required
mt={14}
/>

<Button type="submit" fullWidth mt={14}>
Login
</Button>
</form>
</Modal>
</div>
);
}

export default Login;
